import { Component, OnInit } from '@angular/core';
import { PresetConfigComponent } from '../preset-config/preset-config.component';

@Component({
  selector: 'app-preset-config-page',
  templateUrl: '../preset-config/preset-config.component.html',
  styleUrls: ['../preset-config/preset-config.component.scss']
})
export class PresetConfigPageComponent extends PresetConfigComponent {
  public isPage = true;

  close() {
    this.zone.run(() => {
      this.router.navigateByUrl('/app/config');
    });
  }
}

<div>
  <h3 mat-dialog-title *ngIf="!isPage">Preset Configs</h3>
  <h3 *ngIf="isPage">Preset Configs</h3>
  <div mat-dialog-content [class.overflow-limit]="!isPage">
    <mat-card *ngFor="let preset of presets">
      <mat-card-header>
          <div *ngIf="preset.imageURL" mat-card-avatar style="background-size: cover;" [style.background-image]="'url(' + preset.imageURL + ')'"></div>
        <mat-card-title>{{ preset.name }}</mat-card-title>
        <mat-card-subtitle>{{ preset.category }}</mat-card-subtitle>
      </mat-card-header>
      <mat-card-content>
        <p>{{ preset.description }}</p>
      </mat-card-content>
      <mat-card-actions>
        <button mat-button (click)="addPreset(preset.id)" [disabled]="!creationEnabled">Add Preset</button>
      </mat-card-actions>
    </mat-card>
    <div mat-dialog-actions>
      <button style="margin-top: 10px;" mat-button (click)="close()">Cancel</button>
    </div>
  </div>
</div>
